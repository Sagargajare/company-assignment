üêõ TROUBLESHOOTING
================================================================================

PORT ALREADY IN USE:
  lsof -i :5173   # Check frontend port
  lsof -i :3001   # Check backend port
  lsof -i :5432   # Check database port
  kill -9 <PID>   # Kill the process

SERVICE NOT STARTING:
  ./logs-local.sh                              # View logs
  docker-compose -f docker-compose.local.yml ps # Check status

DATABASE CONNECTION FAILED:
  docker logs traya-postgres-local             # Check postgres logs
  docker-compose -f docker-compose.local.yml restart postgres

CODE CHANGES NOT REFLECTING:
  docker-compose -f docker-compose.local.yml restart backend

NUCLEAR OPTION (clean everything):
  docker-compose -f docker-compose.local.yml down -v --rmi all
  ./start-local.sh
